```python
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib
matplotlib.use('Agg')  # 非交互式后端

# 1. 读取 Excel 文件
excel_path = "your_data.xlsx"  # 替换为您的文件路径
df = pd.read_excel(excel_path)

# 2. 数据分析
print("=== 数据概览 ===")
print(df.head())
print("\n=== 数据统计 ===")
print(df.describe())

# 3. 创建图表
fig, axes = plt.subplots(2, 2, figsize=(12, 10))
fig.suptitle('Excel 数据分析报告', fontsize=16)

# 图1: 柱状图（假设第一列是类别，第二列是数值）
if len(df.columns) >= 2:
    ax1 = axes[0, 0]
    df.plot(kind='bar', x=df.columns[0], y=df.columns[1], ax=ax1, legend=False)
    ax1.set_title('柱状图')
    ax1.tick_params(axis='x', rotation=45)

# 图2: 折线图
if len(df.columns) >= 2:
    ax2 = axes[0, 1]
    df.plot(kind='line', y=df.columns[1], ax=ax2, legend=False)
    ax2.set_title('折线图')

# 图3: 饼图（如果有分类数据）
ax3 = axes[1, 0]
if df[df.columns[0]].dtype == 'object':
    df[df.columns[0]].value_counts().plot(kind='pie', ax=ax3, autopct='%1.1f%%')
    ax3.set_title('饼图')
else:
    ax3.text(0.5, 0.5, '无分类数据', ha='center', va='center')
    ax3.set_title('饼图')

# 图4: 数值分布直方图
ax4 = axes[1, 1]
numeric_cols = df.select_dtypes(include=['number']).columns
if len(numeric_cols) > 0:
    df[numeric_cols[0]].hist(ax=ax4, bins=20)
    ax4.set_title(f'{numeric_cols[0]} 分布')

plt.tight_layout()
plt.savefig('chart.png', dpi=150, bbox_inches='tight')
plt.close()

# 4. 生成 HTML 报告
html_content = f"""



    
    
    Excel 数据分析报告
    


    
📊 Excel 数据分析报告

    
    
📈 数据概览

    

        

            
{len(df)}

            
总行数

        

        

            
{len(df.columns)}

            
总列数

        

        

            
{df.isnull().sum().sum()}

            
缺失值数量

        

    

    
    
📊 图表分析

    

        
    

    
    
📋 数据表格（前20行）

    {df.head(20).to_html(index=False, classes='data-table')}
    
    
📉 统计摘要

    {df.describe().to_html(classes='data-table')}
    
    

        
报告生成时间: {pd.Timestamp.now().strftime('%Y-%m-%d %H:%M:%S')}

    



"""

# 5. 保存 HTML 文件
with open('report.html', 'w', encoding='utf-8') as f:
    f.write(html_content)

print("\n✅ 分析完成！")
print("📊 图表已保存: chart.png")
print("📄 HTML报告已保存: report.html")
```
