from reportlab.lib.pagesizes import A4
from reportlab.pdfgen import canvas
from reportlab.lib.colors import Color, HexColor
from reportlab.pdfbase import pdfmetrics
from reportlab.pdfbase.ttfonts import TTFont
from reportlab.lib.units import cm
import os

# 创建PDF
output_path = "lobster_poster.pdf"
c = canvas.Canvas(output_path, pagesize=A4)
width, height = A4

# 定义颜色
deep_red = HexColor('#8B0000')
gold = HexColor('#FFD700')
orange = HexColor('#FF6B35')
cream = HexColor('#FFF8DC')
dark_blue = HexColor('#1a3a5c')

# 背景渐变效果 - 深蓝到深红
for i in range(int(height)):
    ratio = i / height
    r = int(26 + (139 - 26) * ratio)
    g = int(58 + (0 - 58) * ratio)
    b = int(92 + (0 - 92) * ratio)
    c.setStrokeColor(Color(r/255, g/255, b/255))
    c.setFillColor(Color(r/255, g/255, b/255))
    c.line(0, i, width, i)

# 装饰性边框
c.setStrokeColor(gold)
c.setLineWidth(3)
c.roundRect(15, 15, width-30, height-30, 10, stroke=1, fill=0)

c.setLineWidth(1.5)
c.roundRect(25, 25, width-50, height-50, 8, stroke=1, fill=0)

# 顶部装饰横条
c.setFillColor(gold)
c.rect(40, height-120, width-80, 3, fill=1, stroke=0)

# 主标题区域
c.setFillColor(gold)
c.setFont("Helvetica-Bold", 72)
c.drawCentredString(width/2, height-180, "LOBSTER")

c.setFont("Helvetica-Bold", 36)
c.drawCentredString(width/2, height-230, "FRESH & DELICIOUS")

# 龙虾图标区域 (用文字和形状模拟)
c.setFillColor(orange)
c.setFont("Helvetica-Bold", 120)
c.drawCentredString(width/2, height-380, "🦞")

# 特色文字
c.setFillColor(cream)
c.setFont("Helvetica-Bold", 28)
c.drawCentredString(width/2, height-480, "Premium Quality")

c.setFont("Helvetica", 18)
c.drawCentredString(width/2, height-510, "Freshly Caught • Sustainably Sourced")

# 分隔线
c.setStrokeColor(gold)
c.setLineWidth(2)
c.line(100, height-540, width-100, height-540)

# 特色卖点
features = [
    "🌊  Ocean Fresh Daily",
    "🍽️  Premium Selection", 
    "👨‍🍳  Chef's Special Recipes",
    "🚚  Fast Delivery Available"
]

c.setFillColor(cream)
c.setFont("Helvetica", 20)
y_pos = height - 590
for feature in features:
    c.drawCentredString(width/2, y_pos, feature)
    y_pos -= 35

# 促销信息框
c.setFillColor(orange)
c.roundRect(80, 140, width-160, 100, 15, fill=1, stroke=0)

c.setFillColor(HexColor('#FFFFFF'))
c.setFont("Helvetica-Bold", 32)
c.drawCentredString(width/2, 205, "SPECIAL OFFER")

c.setFont("Helvetica-Bold", 24)
c.drawCentredString(width/2, 165, "20% OFF First Order!")

# 底部联系信息
c.setFillColor(gold)
c.setFont("Helvetica", 14)
c.drawCentredString(width/2, 100, "Order Now: www.lobster-fresh.com")
c.drawCentredString(width/2, 80, "Tel: 400-888-8888 | WeChat: LobsterFresh")

# 底部装饰横条
c.rect(40, 55, width-80, 3, fill=1, stroke=0)

# 角落装饰
def draw_corner_decoration(x, y, size, rotation):
    c.saveState()
    c.translate(x, y)
    c.rotate(rotation)
    c.setStrokeColor(gold)
    c.setLineWidth(2)
    c.line(0, 0, size, 0)
    c.line(0, 0, 0, size)
    c.restoreState()

draw_corner_decoration(35, height-35, 30, 0)
draw_corner_decoration(width-35, height-35, 30, -90)
draw_corner_decoration(35, 35, 30, 90)
draw_corner_decoration(width-35, 35, 30, 180)

# 保存PDF
c.save()

print(f"海报已生成: {output_path}")
print(f"文件大小: {os.path.getsize(output_path)} bytes")
